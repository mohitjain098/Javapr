7. Java Program to Get and Set State and Get Label of a Check Box

Solution

        JFrame frame = new JFrame("Check Box Example");
        JCheckBox checkBox = new JCheckBox("Check me!");
        JLabel label = new JLabel("Checkbox is not selected.");

9. Java Program to Create and Switch between Frames using Buttons

Solution


import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class FrameSwitchExample {
    public static void main(String[] args) {
        JFrame frame1 = new JFrame("Frame 1");
        JButton switchButton = new JButton("Switch to Frame 2");
        JFrame frame2 = new JFrame("Frame 2");
        JButton switchButton2 = new JButton("Switch to Frame 1");

        switchButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame1.setVisible(false);
                frame2.setVisible(true);
            }
        });

        switchButton2.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame2.setVisible(false);
                frame1.setVisible(true);
            }
        });

        frame1.add(switchButton);
        frame1.setSize(300, 200);
        frame1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame1.setVisible(true);
        
        frame2.add(switchButton2);
        frame2.setSize(300, 200);
        frame2.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame2.setVisible(false);
    }
}
Output: Two frames with buttons to switch between them.

14. Java Program to Check Which Fonts are Available in a Local System

Solution


import javax.swing.*;

public class AvailableFonts {
    public static void main(String[] args) {
        String[] fonts = GraphicsEnvironment.getLocalGraphicsEnvironment().getAvailableFontFamilyNames();
        JFrame frame = new JFrame("Available Fonts");
        JTextArea textArea = new JTextArea(10, 30);

        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}

18. Java Program to Display Several Dots on the Screen Continuously

Solution


import javax.swing.*;
import java.awt.*;

public class DotsDisplay extends JPanel {
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        g.setColor(Color.RED);
        for (int i = 0; i < 100; i++) {
            g.fillOval((int)(Math.random() * getWidth()), (int)(Math.random() * getHeight()), 10, 10);
        }
    }

    public static void main(String[] args) {
        JFrame frame = new JFrame("Dots Display");
        DotsDisplay panel = new DotsDisplay();
        frame.add(panel);
        frame.setSize(400, 300);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}
Output: A frame displaying random dots continuously.

19. Java Program to Validate the TextField for only entering numbers using KeyListener

Solution


import javax.swing.*;
import java.awt.event.KeyEvent;
import java.awt.event.KeyListener;

public class NumberValidation {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Number Validation");
        JTextField textField = new JTextField(10);
        JLabel label = new JLabel("Enter numbers only:");

        textField.addKeyListener(new KeyListener() {
            @Override
            public void keyTyped(KeyEvent e) {
                char c = e.getKeyChar();
                if (!Character.isDigit(c)) {
                    e.consume(); // ignore the event
                }
            }
            @Override
            public void keyPressed(KeyEvent e) {}
            @Override
            public void keyReleased(KeyEvent e) {}
        });

        frame.setLayout(new BoxLayout(frame.getContentPane(), BoxLayout.Y_AXIS));
        frame.add(label);
        frame.add(textField);
        frame.setSize(300, 200);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}
Output: A text field that only accepts numeric input.

20. Java Program to Validate the TextField for only entering email id using KeyAdapter

Solution


import javax.swing.*;
import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;

public class EmailValidation {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Email Validation");
        JTextField textField = new JTextField(20);
        JLabel label = new JLabel("Enter email id:");

        textField.addKeyListener(new KeyAdapter() {
            @Override
            public void keyTyped(KeyEvent e) {
                char c = e.getKeyChar();
                if (!(Character.isLetterOrDigit(c) || c == '@' || c == '.' || c == '_')) {
                    e.consume(); // ignore the event
                }
            }
        });

        frame.setLayout(new BoxLayout(frame.getContentPane(), BoxLayout.Y_AXIS));
        frame.add(label);
        frame.add(textField);
        frame.setSize(300, 200);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}
Output: A text field that only accepts valid email characters.
